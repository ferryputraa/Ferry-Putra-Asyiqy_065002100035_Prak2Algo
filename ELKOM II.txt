a = int(input('Masukkan angka pertama : '))

c = int(input('Masukkan angka ketiga  : '))

    print('Angka terbesar adalah  :',a)

    print('Angka terbesar adalah  :',b)
else :
    print('Angka terbesar adalah  :',c)
